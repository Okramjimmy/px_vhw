        function convertPxToVhVw() {

            const pxValue = parseFloat(document.getElementById('pxInput').value);


            // Check if the input value is a valid number

            if (isNaN(pxValue) || pxValue <= 0) {

                alert('Please enter a valid positive number');

                return;

            }


            // Get the viewport dimensions

            const viewportWidth = window.innerWidth;

            const viewportHeight = window.innerHeight;


            // Calculate the corresponding vh and vw values

            const vhValue = (pxValue / viewportHeight) * 100;

            const vwValue = (pxValue / viewportWidth) * 100;


            // Display the results

            document.getElementById('vhResult').textContent = `${pxValue}px = ${vhValue.toFixed(2)}vh`;

            document.getElementById('vwResult').textContent = `${pxValue}px = ${vwValue.toFixed(2)}vw`;

        }


        // Optional: Update the results live when the window size changes

        window.addEventListener('resize', function() {

            const pxValue = parseFloat(document.getElementById('pxInput').value);

            if (!isNaN(pxValue) && pxValue > 0) {

                convertPxToVhVw(); // Recalculate and update the result on resize

            }

        });
